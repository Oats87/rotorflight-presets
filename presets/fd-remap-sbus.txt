#$ TITLE: FlyDragon F722V2 FBus Servo 5 Remap
#$ FIRMWARE_VERSION: 4.4
#$ BOARD_NAME: FLYDRAGONF722_V2
#$ CATEGORY: REMAPPING
#$ STATUS: COMMUNITY
#$ KEYWORDS: remap, fbus
#$ AUTHOR: FlyDragonRC
#$ DESCRIPTION: Remaps the FBUS port to Servo 5
#$ PRIORITY: 0

resource SERIAL_TX 2 none
resource SERVO 5 A02
timer A02 AF3
mixer input CH10 -1000 1000 1000
mixer rule 10 set CH10 S5 1000 0
